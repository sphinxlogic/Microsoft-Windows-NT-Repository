INVOKING THE VDE UTILITY AT DCL LEVEL
  $ VDE $ VDE vde-command; vde-command $ HELP VDE
  VDE> vde-command $ VDE/KEPT/LIBRARY=vde-lib-name $ HELP VSC
  VDE> EXIT $ VDE/INTERFACE=DECWINDOWS VDE> HELP
         
FETCHING, RESERVING AND REPLACING MODULES
  FETCH /STREAM=stream [fac]mod.typ
    /GENERATION=3a2b1 /OUTPUT=filespec /STREAM=stream
         
  RESERVE /STREAM=stream [fac]mod.typ "remark"
    /IDENTIFICATION=res-id /OUTPUT=filespec /QUEUE
    /NOCONCURRENT /OVERRIDE=CONCURRENT /REMARK="string"
    /NOPROPAGATE /PROPAGATE=stream /SESSION=session
         
  REPLACE /STREAM=stream [fac]mod.typ "remark"
    /FOLD=fold-id /NOINFORMATION /REPLACEMENT=repl-name
    /IDENTIFICATION=res-id /PROPAGATE=stream /REVIEWER=(user,...)
    /INFORMATION=filespec /NOPROPAGATE /STREAM=stream
    /INPUT=filespec /REMARK="string" /SESSION=session
    /NOTE=n.m /SCT=filespec /REASON=reasoncode
         
  UNRESERVE /STREAM=stream [fac]mod.typ
    /IDENTIFICATION=res-id /STREAM=stream  
  CREATE MODULE[fac]mod.typ
  DELETE MODULE[fac]mod.typ
         
DISPLAYING INFORMATION 1
  SHOW FACILITITY fac-name
  SHOW FOLD[fac]mod.typ
    /IDENTIFICATION=fold-id /STREAM=stream  
         
  SHOW GENERATION[fac]mod.typ;n
    /BEFORE=date-time /IF_DIFFERENT=stream /STREAM=stream
    /SINCE=date-time /IF_SAME=stream /TREE
    /GENERATION=3a2 /STATISTICS=(...) /USERNAME=user
  DIFFERENCES[fac]mod.typ;n [fac]mod.typ;n
  DIFFERENCES[fac]mod.typ;n
    /GENERATION=3a2 /OUTPUT=filespec /STREAM=stream
         
  SHOW MODULE[fac]mod.typ
  SHOW REPLACEMENT repl-name
    /ACCEPTED /NOT_REVIEWED /STREAM=stream
    /REJECTED /BEFORE=date-time /REVIEWER=user
    /IN_REVIEW /SINCE=date-time /USERNAME=user
DISPLAYING INFORMATION 1, cont.
  SHOW RESERVATION[fac]mod.typ
    /BEFORE=date-time /SESSION=session /SINCE=date-time
    /STREAM=stream /USERNAME=user  
  SHOW STREAM stream
    /OPEN /CLOSED /SUCCESSOR
  SHOW USER user
         
CANCELLING FOLD RECORDS
  CANCEL FOLD /STREAM=stream [fac]mod.typ
    /IDENTIFICATION=fold-id /STREAM=stream  
  REPLACE /FOLD /STREAM=stream [fac]mod.typ "remark"
         
REVIEWING REPLACEMENTS
  REVIEW REPLACEMENT repl-name
    NEXT BACK CURRENT FIRST LAST DIRECTORY EXIT
    SELECT [fac]mod.typ ACCEPT "remark" REJECT "remark"
  ACCEPT REPLACEMENT repl-name "remark"
  REJECT REPLACEMENT repl-name "remark"
         
PROJECT LEADER OPERATIONS
  PERFORM REPLACEMENT repl-name /STREAM=stream
  CANCEL REPLACEMENT repl-name /STREAM=stream
         
  INSERT GENERATION[fac]mod.typ /STREAM=orig-stream dest-stream
    /GENERATION=3a2 /STREAM=stream  
         
  MODIFY GENERATION[fac]mod.typ /[NO]OBSOLETE
  REMOVE GENERATION[fac]mod.typ /STREAM=stream
  DELETE GENERATION[fac]mod.typ /STREAM=stream
         
  CREATE STREAM stream /PARENT=stream /CMS_CLASS /REMARK="string"
  MODIFY STREAM stream
    /REVIEWER=(user,...) /SUCCESSOR=stream /OPEN
    /NOREVIEWER=(user,...) /OWNER=user /FROZEN
    /NOTIFY=(user,...) /NAME=new-name /CLOSED
    /NONOTIFY=(user,...) /REMARK="string" /DELETE
  POPULATE STREAM stream /FROM_STREAM=stream
         
LIBRARY MAINTENANCE COMMANDS
  CREATE USER user
  MODIFY USER user
  CREATE LIBRARY lib-name
  MODIFY LIBRARY lib-name
         
CONVENIENCE FEATURES
  @filespec SET FACILITY DEFINE/KEY SET LOG
  HELP SET STREAM SET KEY CANCEL LOG
  SPAWN SHOW DEFAULTS SHOW KEY SET VERIFY
  ATTACH SET EDITOR DEFINE/COMMAND CANCEL VERIFY
  SET PRIV SHOW EDITOR SHOW SYMBOL EXIT /KEPT_PROCESS
1All SHOW commands accept /BRIEF, /FULL, and /TOTAL; some accept /OUTPUT=filespec
