% Add the METAFONT logo font to \fourteenpt etc.
%
%
% This is a heavyhanded approach.  I tried to find a csname
% associated with font family changes that was simple enough
% to be used for an \ifx comparison, but finally gave up.
% The definitions of \fourteenpt etc. are unchanged except
% for the addition of \def\@MF{\umpteenlogo} at the
% end of the visibles.  No provision is made for invisible
% METAFONT, but it wouldn't be difficult.
% \MF is hardwired to produce METAFONT in whatever is the
% \@currsize
%
% Input this file when making a new load of splain.tex
% You will be asked for local modifications; just respond
% \input slogofonts \dump
%

%         SIZE: \normalsize \small \footnotesize \scriptsize \tiny
%               \large \Large \LARGE \huge \Huge
% ( In SliTeX, huge and Huge are mapped into LARGE for all but lcmss )

\catcode`@=11 % Protect local macros.

\font\seventeenlogo   = logo8  scaled \magstep 4 % METAFONT (\small)

\font\twentyninelogo   = logo8  scaled 3583 % METAFONT (\Large)
\font\thirtyfourlogo   = logo8  scaled 4300 % METAFONT (\LARGE)
